node *__fastcall Register_node(node *prev_node, node *new_node)
{
  node *result; // rax
  struct node *mid_node; // [rsp+8h] [rbp-18h]

  mid_node = malloc(0x10uLL);
  mid_node->A = new_node;
  mid_node->B = 0LL;
  if ( prev_node->B )
    *(prev_node->B + 8) = mid_node;
  else
    prev_node->A = mid_node;
  result = prev_node;
  prev_node->B = mid_node;
  return result;
}
